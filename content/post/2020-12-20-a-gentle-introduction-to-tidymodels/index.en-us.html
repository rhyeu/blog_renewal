#### step_center, step_scale는 왜 저렇게 쓰는가?
iris_recipe <- training(iris_split) %>% 
  recipe(Species ~.) %>% # 전처리 시행
  step_corr(all_predictors()) %>% 
  step_center(all_predictors(), -all_outcomes()) %>% 
  step_scale(all_predictors(), -all_outcomes()) %>% 
  prep() # 전처리 시행
